#### 구현예시

<script>
    alert('hello javascript')
    var myFavoriteNum = window.prompt('좋아하는 숫자를 입력하시오');

        name : 'unani',
        age : 27,
        nationality : 'Korea'
    };
    console.log(person['name'])

</script>
